# Template Configuration

```ts title="Type"
interface TemplateConfig {
	answers: TemplateConfigAnswers;
	opts: TemplateConfigOptions;
}
```

## Answers

```ts title="Type"
interface TemplateConfigAnswers {
	[name: string]: any;
}
```

Object containing answers to prompts.

```json title=".tpsrc"
{
	"some-template": {
		// highlight-start
		"answers": {
			"<prompt-name>": "<prompt-value>"
		}
		// highlight-end
	}
}
```

<Example>

```json title=".tpsrc"
{
	"react-component": {
		// highlight-start
		"answers": {
			"css": true
		}
		// highlight-end
	}
}
```

</Example>

## Options

```ts title="Type"
interface TemplateConfigOptions {
	[name: string]: any;
}
```

Object containing options for templates. See
[Template Option API](../template#options) for template options.

```json title=".tpsrc"
{
	"some-template": {
		// highlight-start
		"opts": {
			"<option-name>": "<option-value>"
		}
		// highlight-end
	}
}
```

<Example>

```json title=".tpsrc"
{
	"react-component": {
		// highlight-start
		"opts": {
			"wipe": true
		}
		// highlight-end
	}
}
```

</Example>
